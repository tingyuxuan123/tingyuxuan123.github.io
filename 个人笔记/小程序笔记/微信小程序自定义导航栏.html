# 新建一个 nav-bar 组件

wxml 代码

<!--components/nav-bar/nav-bar.wxml-->
<view class="nav-bar" style="color: {{fontColor}};">
    <view class="status" style="height: {{statusHeight}}px;"></view>
    <view class="nav">
      <view class="left">
        <view class="slot">
          <slot name="left"></slot>
        </view>
        <view class="default iamge">
            <image class="arrow-left" src="/assets/images/icons/arrow_left.png" mode=""/>
        </view>
      </view>
    <view class="center">
      <view class="slot">
        <slot name="center"></slot>
      </view>

</view>

js

// components/nav-bar/nav-bar.js

Component({
  options: {
    /* 如果要是用多插槽的时候 multipleSlots要为true */
    multipleSlots: true
  },
  properties: {
    /* 插槽内容的颜色 */
    fontColor: {
      type: String,
      value: '#ffffff'
    },
    /* 不使用插槽时中间显示的标题 */
    title: {
      type: String,
      value: '导航标题'
    }
  },

  /**
   * 组件的初始数据
   */
  data: {
    /*默认状态栏的高度*/
    statusHeight: 20
  },
  //组件生命名周期
  lifetimes: {
    attached() {
      //获取系统的statusbar的高度
      wx.getSystemInfo().then((res) => {
        this.setData({ statusHeight: res.statusBarHeight })
      })
    }
  }
})

# 使用自定义状态栏

关闭要使用页面的状态栏

{
  "navigationStyle": "custom", //关闭默认导航
  "usingComponents": {
    "nav-bar": "/components/nav-bar/nav-bar" //注册组件
  }
}
